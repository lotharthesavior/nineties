use crate::helpers::load_template;
use actix_web::{get, web, HttpResponse, Responder};
use crate::AppState;

#[get("/")]
pub async fn home(data: web::Data) -> impl Responder {
    HttpResponse::Ok().body(load_template("home.html", vec![("name", &data.app_name)]))
}
